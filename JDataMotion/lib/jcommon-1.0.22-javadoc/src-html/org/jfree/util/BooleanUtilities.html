044package org.jfree.util;
045
046/**
047 * Utility methods for working with <code>Boolean</code> objects.
048 * 
049 * @author David Gilbert
050 */
051public class BooleanUtilities {
052
053    /**
054     * Private constructor prevents object creation.
055     */
056    private BooleanUtilities() {
057    }
058
059    /**
060     * Returns the object equivalent of the boolean primitive.
061     * <p>
062     * A similar method is provided by the Boolean class in JDK 1.4, but you can use this one
063     * to remain compatible with earlier JDKs.
064     * 
065     * @param b  the boolean value.
066     * 
067     * @return <code>Boolean.TRUE</code> or <code>Boolean.FALSE</code>.
068     */
069    public static Boolean valueOf(final boolean b) {
070        return (b ? Boolean.TRUE : Boolean.FALSE);
071    }
072    
073}
