046package org.jfree.ui.about;
047
048import java.awt.BorderLayout;
049import java.util.List;
050import javax.swing.JPanel;
051import javax.swing.JScrollPane;
052import javax.swing.JTable;
053import javax.swing.table.TableModel;
054
055/**
056 * A panel containing a table that lists the contributors to a project.
057 * <P>
058 * Used in the AboutFrame class.
059 *
060 * @author David Gilbert
061 */
062public class ContributorsPanel extends JPanel {
063
064    /** The table. */
065    private JTable table;
066
067    /** The data. */
068    private TableModel model;
069
070    /**
071     * Creates a new contributors panel.
072     *
073     * @param contributors  a list of contributors (represented by Contributor objects).
074     */
075    public ContributorsPanel(final List contributors) {
076
077        setLayout(new BorderLayout());
078        this.model = new ContributorsTableModel(contributors);
079        this.table = new JTable(this.model);
080        add(new JScrollPane(this.table));
081
082    }
083
084}
