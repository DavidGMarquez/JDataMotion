044package org.jfree.xml.factory.objects;
045
046import java.awt.geom.Rectangle2D;
047
048/**
049 * An object-description for a <code>Rectangle2D</code> object.
050 *
051 * @author Thomas Morgner
052 */
053public class Rectangle2DObjectDescription extends AbstractObjectDescription {
054
055    /**
056     * Creates a new object description.
057     */
058    public Rectangle2DObjectDescription() {
059        super(Rectangle2D.class);
060        setParameterDefinition("width", Float.class);
061        setParameterDefinition("height", Float.class);
062        setParameterDefinition("x", Float.class);
063        setParameterDefinition("y", Float.class);
064    }
065
066    /**
067     * Creates an object based on this description.
068     *
069     * @return The object.
070     */
071    public Object createObject() {
072        final Rectangle2D rect = new Rectangle2D.Float();
073
074        final float w = getFloatParameter("width");
075        final float h = getFloatParameter("height");
076        final float x = getFloatParameter("x");
077        final float y = getFloatParameter("y");
078        rect.setRect(x, y, w, h);
079        return rect;
080    }
081
082    /**
083     * Returns a parameter value as a float.
084     *
085     * @param param  the parameter name.
086     *
087     * @return The float value.
088     */
089    private float getFloatParameter(final String param) {
090        final Float p = (Float) getParameter(param);
091        if (p == null) {
092            return 0;
093        }
094        return p.floatValue();
095    }
096
097    /**
098     * Sets the parameters of this description object to match the supplied object.
099     *
100     * @param o  the object (should be an instance of <code>Rectangle2D</code>).
101     *
102     * @throws ObjectFactoryException if the object is not an instance of <code>Rectangle2D</code>.
103     */
104    public void setParameterFromObject(final Object o) throws ObjectFactoryException {
105        if (!(o instanceof Rectangle2D)) {
106            throw new ObjectFactoryException("The given object is no java.awt.geom.Rectangle2D.");
107        }
108
109        final Rectangle2D rect = (Rectangle2D) o;
110        final float x = (float) rect.getX();
111        final float y = (float) rect.getY();
112        final float w = (float) rect.getWidth();
113        final float h = (float) rect.getHeight();
114
115        setParameter("x", new Float(x));
116        setParameter("y", new Float(y));
117        setParameter("width", new Float(w));
118        setParameter("height", new Float(h));
119    }
120
121}
