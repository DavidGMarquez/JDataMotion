043package org.jfree.xml.attributehandlers;
044
045/**
046 * A class that handles the conversion of {@link Byte} attributes to and from an appropriate
047 * {@link String} representation.
048 */
049public class ByteAttributeHandler implements AttributeHandler {
050
051    /**
052     * Creates a new attribute handler.
053     */
054    public ByteAttributeHandler() {
055        super();
056    }
057
058    /**
059     * Converts the attribute to a string.
060     * 
061     * @param o  the attribute ({@link Byte} expected).
062     * 
063     * @return A string representing the {@link Byte} value.
064     */
065    public String toAttributeValue(final Object o) {
066        final Byte in = (Byte) o;
067        return in.toString();
068    }
069
070    /**
071     * Converts a string to a {@link Byte}.
072     * 
073     * @param s  the string.
074     * 
075     * @return a {@link Byte}.
076     */
077    public Object toPropertyValue(final String s) {
078        return new Byte(s);
079    }
080    
081}
