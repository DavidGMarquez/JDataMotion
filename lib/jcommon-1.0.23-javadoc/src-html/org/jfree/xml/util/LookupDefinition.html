043package org.jfree.xml.util;
044
045/**
046 * A lookup definition.
047 */
048public class LookupDefinition {
049    
050    /** The property name. */
051    private String propertyName;
052    
053    /** The registry key. */
054    private String registryKey;
055
056    /**
057     * Creates a new lookup definition.
058     * 
059     * @param propertyName  the property name.
060     * @param registryKey  the registry key.
061     */
062    public LookupDefinition(final String propertyName, final String registryKey) {
063        this.propertyName = propertyName;
064        this.registryKey = registryKey;
065    }
066
067    /**
068     * Returns the property name.
069     * 
070     * @return the property name.
071     */
072    public String getPropertyName() {
073        return this.propertyName;
074    }
075
076    /**
077     * Returns the registry key.
078     * 
079     * @return the registry key.
080     */
081    public String getRegistryKey() {
082        return this.registryKey;
083    }
084    
085}
